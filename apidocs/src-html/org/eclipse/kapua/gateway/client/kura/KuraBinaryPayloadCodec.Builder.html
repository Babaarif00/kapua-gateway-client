012package org.eclipse.kapua.gateway.client.kura;
013
014import java.nio.ByteBuffer;
015import java.time.Instant;
016import java.util.Map;
017
018import org.eclipse.kapua.gateway.client.BinaryPayloadCodec;
019import org.eclipse.kapua.gateway.client.Payload;
020import org.eclipse.kapua.gateway.client.kura.payload.KuraPayloadProto;
021import org.eclipse.kapua.gateway.client.kura.payload.KuraPayloadProto.KuraPayload;
022import org.eclipse.kapua.gateway.client.kura.utils.Metrics;
023import org.eclipse.kapua.gateway.client.utils.Buffers;
024
025public class KuraBinaryPayloadCodec implements BinaryPayloadCodec {
026
027    public static class Builder {
028
029        public KuraBinaryPayloadCodec build() {
030            return new KuraBinaryPayloadCodec();
031        }
032    }
033
034    private KuraBinaryPayloadCodec() {
035    }
036
037    @Override
038    public ByteBuffer encode(final Payload payload, final ByteBuffer buffer) throws Exception {
039        final KuraPayloadProto.KuraPayload.Builder builder = KuraPayload.newBuilder();
040        builder.setTimestamp(payload.getTimestamp().toEpochMilli());
041        Metrics.buildMetrics(builder, payload.getValues());
042
043        final byte[] data = builder.build().toByteArray();
044        if (buffer == null) {
045            // create a wrapped buffer
046            return Buffers.wrap(data);
047        } else if (buffer.remaining() < data.length) {
048            // create a new, merged buffer
049            buffer.flip();
050            final ByteBuffer newBuffer = ByteBuffer.allocate(buffer.remaining() + data.length);
051            newBuffer.put(buffer);
052            newBuffer.put(data);
053            return newBuffer;
054        } else {
055            buffer.put(data);
056            return buffer;
057        }
058    }
059
060    @Override
061    public Payload decode(final ByteBuffer buffer) throws Exception {
062        final KuraPayload payload = KuraPayload.parseFrom(Buffers.toByteArray(buffer));
063        final Map<String, Object> values = Metrics.extractMetrics(payload);
064        return Payload.of(Instant.ofEpochMilli(payload.getTimestamp()), values);
065    }
066
067}
