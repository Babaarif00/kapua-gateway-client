049@Component(properties = "OSGI-INF/persistence.properties")
050@Designate(ocd = Configuration.class)
051public class PersistenceServiceImpl implements PersistenceService {
052
053    private static final Logger logger = LoggerFactory.getLogger(PersistenceServiceImpl.class);
054
055    private Client client;
056    private Application app;
057
058    @Override
059    public String getId() {
060        return "kapua";
061    }
062
063    @Override
064    public String getLabel(final Locale locale) {
065        return "Eclipse Kapua";
066    }
067
068    @Override
069    public void store(final Item item) {
070        store(item, null);
071    }
072
073    @Override
074    public void store(final Item item, final String alias) {
075        logger.debug("Store - item: {}, alias: {}", item, alias);
076
077        final String key = alias == null ? item.getName() : alias;
078        if (key == null || key.isEmpty()) {
079            return;
080        }
081
082        final Map<String, Object> data = new HashMap<>();
083        final State state = item.getState();
084        if (state instanceof DecimalType) {
085            data.put("value", ((DecimalType) state).longValue());
086        } else if (state instanceof StringType) {
087            data.put("value", state.toString());
088        } else if (state instanceof OnOffType) {
089            data.put("value", state.equals(OnOffType.ON));
090        } else if (state instanceof UpDownType) {
091            data.put("value", state.equals(UpDownType.UP));
092        } else if (state instanceof PlayPauseType) {
093            data.put("value", state.equals(PlayPauseType.PLAY));
094        } else {
095            return;
096        }
097
098        final Sender<RuntimeException> sender;
099
100        synchronized (this) {
101            if (app == null) {
102                logger.debug("Not configured, discarding event");
103                return;
104            }
105            final Topic topic = Topic.of(key);
106            final Payload payload = Payload.of(data);
107
108            logger.debug("Topic: {}, payload: {}", topic, payload);
109
110            sender = app.data(topic).errors(Errors.ignore());
111            sender.send(payload);
112        }
113    }
114
115    @Activate
116    protected synchronized void activate(final Map<String, ?> properties) throws Exception {
117        if (logger.isWarnEnabled()) {
118            logger.warn("Create new Kapua client: {}", new TreeMap<>(properties));
119        }
120
121        final String broker = getOrDefault(properties, "brokerUrl", null);
122        if (broker == null || broker.isEmpty()) {
123            logger.warn("Missing broker URL. Not starting.");
124            return;
125        }
126
127        final KuraMqttProfile<PahoClient.Builder> profile = newProfile(PahoClient.Builder::new);
128        profile.accountName(getOrDefault(properties, "accoutName", "kapua-sys"));
129        profile.clientId(getOrDefault(properties, "clientId", UUID.randomUUID().toString()));
130        profile.brokerUrl(broker);
131
132        final String user = getOrDefault(properties, "user", null);
133        final String password = getOrDefault(properties, "password", null);
134
135        if (user != null && password != null) {
136            profile.credentials(Credentials.userAndPassword(user, password));
137        }
138
139        this.client = profile.build();
140
141        this.app = client.buildApplication("smarthome").build();
142    }
143
144    private static String getOrDefault(final Map<String, ?> properties, final String key, final String defaultValue) {
145        if (properties == null) {
146            return defaultValue;
147        }
148
149        final Object value = properties.get(key);
150        if (value == null) {
151            return defaultValue;
152        }
153
154        return value.toString();
155    }
156
157    @Deactivate
158    protected synchronized void deactivate() throws Exception {
159        if (client != null) {
160            try {
161                this.client.close();
162            } finally {
163                client = null;
164                app = null;
165            }
166        }
167    }
168
169    @Modified
170    protected synchronized void modified(final Map<String, ?> properties) throws Exception {
171        deactivate();
172        activate(properties);
173    }
174}
